using UnityEngine;
using UnityEngine.UI;
using System.Collections;

public class SimpleFadeIn : MonoBehaviour
{
    [Header("UI Fade")]
    public Canvas fadeCanvas;
    public RawImage fadeImage;
    public float fadeDuration = 2f;
    void Start()
    {
        if (fadeImage != null)
        {
            Color c = fadeImage.color;
            c.a = 1f;
            fadeImage.color = c;
        }
        StartCoroutine(FadeIn());
    }
    IEnumerator FadeIn()
    {
        if (fadeImage == null)
        {
            yield break;
        }
        float t = 0f;
        Color c = fadeImage.color;

        while (t < fadeDuration)
        {
            t+=Time.deltaTime;
            float lerp = t/fadeDuration;
            c.a = Mathf.Lerp(1f, 0f, lerp);
            fadeImage.color = c;
            yield return null;
        }
        c.a = 0f;
        fadeImage.color = c;
        if (fadeCanvas != null)
        {
            fadeCanvas.gameObject.SetActive(true);
        }
    }
    public IEnumerator FadeOut()
    {
        fadeCanvas.gameObject.SetActive(true);
        float t = 0f;
        Color c = fadeImage.color;

        c.a = 0f;
        fadeImage.color = c;

        while (t < fadeDuration)
        {
            t+=Time.deltaTime;
            float lerp = t/fadeDuration;
            c.a = Mathf.Lerp(0f, 1f, lerp);
            fadeImage.color = c;
            yield return null;
        }
        c.a = 1f;
        fadeImage.color = c;
    }
    public IEnumerator FadeInAgain()
    {
        float t = 0f;
        Color c = fadeImage.color;

        while (t < fadeDuration)
        {
            t+=Time.deltaTime;
            float lerp = t/fadeDuration;
            c.a = Mathf.Lerp(1f, 0f, lerp);
            fadeImage.color = c;
            yield return null;
        }
        c.a = 0f;
        fadeImage.color = c;
        fadeCanvas.gameObject.SetActive(false);
    }
}
