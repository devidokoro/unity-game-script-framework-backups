[RequireComponent(typeof(Rigidbody))]
[RequireComponent(typeof(CapsuleCollider))]
public class FirstPersonController : MonoBehaviour
{
    private Rigidbody rb;
    private CapsuleCollider capsuleCollider; 

    #region Camera Movement Variables

    [Tooltip("Drag your Cinemachine Virtual Camera here!")]
    public CinemachineCamera playerCamera; // Converted to Cinemachine

    public float fov = 60f;
    public bool invertCamera = false;
    public bool cameraCanMove = true;
    public float mouseSensitivity = 2f;
    public float maxLookAngle = 50f;

    // Crosshair
    public bool lockCursor = true;
    public bool crosshair = true;
    public Sprite crosshairImage;
    public Color crosshairColor = Color.white;

    // Internal Variables
    private float yaw = 0.0f;
    private float pitch = 0.0f;
    private Image crosshairObject;

    #region Camera Zoom Variables

    public bool enableZoom = true;
    public bool holdToZoom = false;
    public KeyCode zoomKey = KeyCode.Mouse1;
    public float zoomFOV = 30f;
    public float zoomStepTime = 5f;

    // Internal Variables
    private bool isZoomed = false;

    #endregion
    #endregion

    #region Movement Variables

    public bool playerCanMove = true;
    public float walkSpeed = 5f;
    public float maxVelocityChange = 10f;

    // Internal Variables
    private bool isWalking = false;

    #region Sprint

    public bool enableSprint = true;
    public bool unlimitedSprint = false;
    public KeyCode sprintKey = KeyCode.LeftShift;
    public float sprintSpeed = 7f;
    public float sprintDuration = 5f;
    public float sprintCooldown = .5f;
    public float sprintFOV = 80f;
    public float sprintFOVStepTime = 10f;

    // Sprint Bar
    public bool useSprintBar = true;
    public bool hideBarWhenFull = true;
    public Image sprintBarBG;
    public Image sprintBar;
    public float sprintBarWidthPercent = .3f;
    public float sprintBarHeightPercent = .015f;

    // Internal Variables
    private CanvasGroup sprintBarCG;
    private bool isSprinting = false;
    private float sprintRemaining;
    private float sprintBarWidth;
    private float sprintBarHeight;
    private bool isSprintCooldown = false;
    private float sprintCooldownReset;

    #endregion

    #region Jump

    public bool enableJump = true;
    public KeyCode jumpKey = KeyCode.Space;
    public float jumpPower = 5f;

    // Internal Variables
    public bool isGrounded = false;

    #endregion

    #region Crouch

    public bool enableCrouch = true;
    public bool holdToCrouch = true;
    public KeyCode crouchKey = KeyCode.LeftControl;
    public float crouchHeight = .75f;
    public float speedReduction = .5f;

    // Internal Variables
    private bool isCrouched = false;
    private float originalColliderHeight;
    private Vector3 originalCameraJointPos;

    #endregion
    #endregion

    #region Head Bob

    public bool enableHeadBob = true;
    public Transform joint;
    public float bobSpeed = 10f;
    public Vector3 bobAmount = new Vector3(.15f, .05f, 0f);

    // Internal Variables
    private Vector3 jointOriginalPos;
    private float timer = 0;

    #endregion

    #region CamShake
    [Header("Testing Key")]
    public KeyCode testShakeKey = KeyCode.K;
    public float testMagnitude = .2f;
    
    [Header("Shake Audio Settings")]
    public AudioSource shakeAudioSource;
    public AudioClip shakeSoundClip;

    private bool isShaking = false;
    private Vector3 shakeOffset = Vector3.zero;
    public void ActivateShake(float magnitude)
    {
        if (!isShaking) StartCoroutine(Shake(magnitude));
    }
    public void StopShake() {
        isShaking = false;
    }
    private void TestShakeInput()
    {
        if (Input.GetKeyDown(testShakeKey))
        {
            ActivateShake(testMagnitude);
        }
        if (Input.GetKeyUp(testShakeKey))
        {
            StopShake();
        }
    }
    IEnumerator Shake(float magnitude)
    {
        isShaking = true;
        if (shakeAudioSource != null && shakeSoundClip != null)
        {
            shakeAudioSource.clip = shakeSoundClip;
            shakeAudioSource.loop = true;
            shakeAudioSource.Play();
        }
        while (isShaking)
        {
            shakeOffset = Random.insideUnitSphere * magnitude;
            yield return null;
        }
        if (shakeAudioSource != null) shakeAudioSource.Stop();

        float decayDuration = .5f;
        float decayTime = 0f;

        while (decayTime < decayDuration)
        {
            decayTime += Time.deltaTime;
            float t = decayTime / decayDuration;
            float dynamicMagnitude = Mathf.Lerp(magnitude, 0f, t);
            shakeOffset = Random.insideUnitSphere * dynamicMagnitude;
            yield return null;
        }
        shakeOffset = Vector3.zero;
    }
    #endregion

    private void Awake()
    {
        rb = GetComponent<Rigidbody>();
        capsuleCollider = GetComponent<CapsuleCollider>();

        // Fallback: search for Cinemachine Camera in children if slot left empty
        if (playerCamera == null)
        {
            playerCamera = GetComponentInChildren<CinemachineCamera>();
        }

        crosshairObject = GetComponentInChildren<Image>();

        if (playerCamera != null)
        {
            playerCamera.Lens.FieldOfView = fov; // Updated for Cinemachine
        }
        
        if (capsuleCollider != null) originalColliderHeight = capsuleCollider.height;
        if (joint != null) jointOriginalPos = joint.localPosition;

        if (!unlimitedSprint)
        {
            sprintRemaining = sprintDuration;
            sprintCooldownReset = sprintCooldown;
        }
    }

    void Start()
    {
        if(lockCursor)
        {
            Cursor.lockState = CursorLockMode.Locked;
        }

        if(crosshair && crosshairObject != null)
        {
            crosshairObject.sprite = crosshairImage;
            crosshairObject.color = crosshairColor;
        }
        else if (crosshairObject != null)
        {
            crosshairObject.gameObject.SetActive(false);
        }

        #region Sprint Bar

        sprintBarCG = GetComponentInChildren<CanvasGroup>();

        if(useSprintBar)
        {
            if(sprintBarBG != null) sprintBarBG.gameObject.SetActive(true);
            if(sprintBar != null) sprintBar.gameObject.SetActive(true);

            float screenWidth = Screen.width;
            float screenHeight = Screen.height;

            sprintBarWidth = screenWidth * sprintBarWidthPercent;
            sprintBarHeight = screenHeight * sprintBarHeightPercent;

            if (sprintBarBG != null) sprintBarBG.rectTransform.sizeDelta = new Vector3(sprintBarWidth, sprintBarHeight, 0f);
            if (sprintBar != null) sprintBar.rectTransform.sizeDelta = new Vector3(sprintBarWidth - 2, sprintBarHeight - 2, 0f);

            if(hideBarWhenFull && sprintBarCG != null)
            {
                sprintBarCG.alpha = 0;
            }
        }
        else
        {
            if (sprintBarBG != null) sprintBarBG.gameObject.SetActive(false);
            if (sprintBar != null) sprintBar.gameObject.SetActive(false);
        }

        #endregion
    }

    private void Update()
    {
        #region Camera Movement
        if(cameraCanMove && playerCamera != null)
        {
            yaw = transform.localEulerAngles.y + Input.GetAxis("Mouse X") * mouseSensitivity;

            if (!invertCamera)
            {
                pitch -= mouseSensitivity * Input.GetAxis("Mouse Y");
            }
            else
            {
                pitch += mouseSensitivity * Input.GetAxis("Mouse Y");
            }

            pitch = Mathf.Clamp(pitch, -maxLookAngle, maxLookAngle);

            transform.localEulerAngles = new Vector3(0, yaw, 0);
            if (joint != null)
            {
                joint.localEulerAngles = new Vector3(pitch, 0, 0);
            } 
            else
            {
                playerCamera.transform.localEulerAngles = new Vector3(pitch, 0, 0);
            }
        }
        #endregion

        #region Camera Zoom (Cinemachine Driver)

        if (enableZoom && playerCamera != null)
        {
            if (holdToZoom)
            {
                isZoomed = Input.GetKey(zoomKey);
            } 
            else
            {
                if(Input.GetKeyDown(zoomKey))
                {
                    isZoomed = !isZoomed;
                }
            }

            // Driven via Cinemachine Lens parameters instead of Camera components directly
            if (isZoomed)
            {
                playerCamera.Lens.FieldOfView = Mathf.Lerp(playerCamera.Lens.FieldOfView, zoomFOV, zoomStepTime * Time.deltaTime);
            } 
            else if (isSprinting)
            {
                playerCamera.Lens.FieldOfView = Mathf.Lerp(playerCamera.Lens.FieldOfView, sprintFOV, sprintFOVStepTime * Time.deltaTime);
            }
            else
            {
                playerCamera.Lens.FieldOfView = Mathf.Lerp(playerCamera.Lens.FieldOfView, fov, zoomStepTime * Time.deltaTime);
            }
        }

        #endregion

        #region Sprint Logic

        if(enableSprint)
        {
            bool moving = Mathf.Abs(Input.GetAxis("Horizontal")) > 0.1f || Mathf.Abs(Input.GetAxis("Vertical")) > 0.1f; 
            
            if (playerCanMove && moving && Input.GetKey(sprintKey) && sprintRemaining > 0f && !isSprintCooldown)
            {
                isSprinting = true;
                isZoomed = false; 

                if (isCrouched) Crouch();
            } 
            else
            {
                isSprinting = false;
            }

            if (isSprinting && playerCamera != null)
            {
                if (!unlimitedSprint)
                {
                    sprintRemaining -= Time.deltaTime; 
                    if (sprintRemaining <= 0)
                    {
                        sprintRemaining = 0;
                        isSprinting = false;
                        isSprintCooldown = true;
                    }
                }
            } 
            else
            {
                if (!unlimitedSprint)
                {
                    sprintRemaining = Mathf.Clamp(sprintRemaining + Time.deltaTime, 0, sprintDuration);
                }
            }

            if (isSprintCooldown)
            {
                sprintCooldown -= Time.deltaTime;
                if (sprintCooldown <= 0)
                {
                    isSprintCooldown = false;
                    sprintCooldown = sprintCooldownReset;
                }
            }

            if (useSprintBar && !unlimitedSprint && sprintBar != null)
            {
                float sprintRemainingPercent = sprintRemaining / sprintDuration;
                sprintBar.transform.localScale = new Vector3(sprintRemainingPercent, 1f, 1f); 

                if (hideBarWhenFull && sprintBarCG != null)
                {
                    if (sprintRemaining >= sprintDuration)
                    {
                        sprintBarCG.alpha = Mathf.Max(0, sprintBarCG.alpha - 3 * Time.deltaTime);
                    } 
                    else
                    {
                        sprintBarCG.alpha = Mathf.Min(1, sprintBarCG.alpha + 5 * Time.deltaTime);
                    }
                }
            }
        }

        #endregion

        #region Jump Triggers

        if(enableJump && Input.GetKeyDown(jumpKey) && isGrounded)
        {
            Jump();
        }

        #endregion

        #region Crouch Triggers

        if (enableCrouch)
        {
            if(Input.GetKeyDown(crouchKey))
            {
                if(holdToCrouch)
                {
                    isCrouched = false;
                    Crouch();
                }
                else
                {
                    Crouch();
                }
            }
            else if(Input.GetKeyUp(crouchKey) && holdToCrouch)
            {
                isCrouched = true;
                Crouch();
            }
        }

        #endregion

        CheckGround();

        if(enableHeadBob)
        {
            HeadBob();
        }

        TestShakeInput();
    }

    void FixedUpdate()
    {
        #region Movement Execution
        if (playerCanMove)
        {
            Vector3 targetVelocity = new Vector3(Input.GetAxis("Horizontal"), 0, Input.GetAxis("Vertical"));

            if ((targetVelocity.x != 0 || targetVelocity.z != 0) && isGrounded)
            {
                isWalking = true;
            }
            else
            {
                isWalking = false;
            }

            float currentSpeed = isSprinting ? sprintSpeed : walkSpeed;
            targetVelocity = transform.TransformDirection(targetVelocity) * currentSpeed;

            Vector3 velocity = rb.linearVelocity;
            Vector3 velocityChange = (targetVelocity - velocity);
            velocityChange.x = Mathf.Clamp(velocityChange.x, -maxVelocityChange, maxVelocityChange);
            velocityChange.z = Mathf.Clamp(velocityChange.z, -maxVelocityChange, maxVelocityChange);
            velocityChange.y = 0;

            rb.AddForce(velocityChange, ForceMode.VelocityChange);
        } 
        #endregion
    }

    private void CheckGround()
    {
        float currentColliderHeight = capsuleCollider != null ? capsuleCollider.height : originalColliderHeight;
        Vector3 origin = transform.position + Vector3.up * 0.1f; 
        Vector3 direction = Vector3.down;
        float distance = (currentColliderHeight * 0.5f) + 0.2f;

        if (Physics.Raycast(origin, direction, out RaycastHit hit, distance))
        {
            Debug.DrawRay(origin, direction * distance, Color.green);
            isGrounded = true;
        }
        else
        {
            Debug.DrawRay(origin, direction * distance, Color.red);
            isGrounded = false;
        }
    }

    private void Jump()
    {
        if (isGrounded)
        {
            rb.linearVelocity = new Vector3(rb.linearVelocity.x, 0f, rb.linearVelocity.z); 
            rb.AddForce(Vector3.up * jumpPower, ForceMode.Impulse);
            isGrounded = false;
        }

        if(isCrouched && !holdToCrouch)
        {
            Crouch();
        }
    }

    private void Crouch()
    {
        if(isCrouched)
        {
            if (capsuleCollider != null) capsuleCollider.height = originalColliderHeight;
            walkSpeed /= speedReduction;

            if (joint != null) 
            {
                joint.localScale = Vector3.one;
                joint.localPosition = jointOriginalPos;
            }

            isCrouched = false;
        }
        else
        {
            if (capsuleCollider != null) capsuleCollider.height = originalColliderHeight * crouchHeight;
            walkSpeed *= speedReduction;

            if (joint != null) 
            {
                joint.localScale = Vector3.one; 
                joint.localPosition = new Vector3(jointOriginalPos.x, jointOriginalPos.y * crouchHeight, jointOriginalPos.z);
            }

            isCrouched = true;
        }
    }

    private void HeadBob()
    {
        if(joint == null) return;

        if(isWalking && isGrounded)
        {
            if(isSprinting)
            {
                timer += Time.deltaTime * (bobSpeed + sprintSpeed);
            }
            else if (isCrouched)
            {
                timer += Time.deltaTime * (bobSpeed * speedReduction);
            }
            else
            {
                timer += Time.deltaTime * bobSpeed;
            }
            
            Vector3 currentBase = isCrouched ? new Vector3(jointOriginalPos.x, jointOriginalPos.y * crouchHeight, jointOriginalPos.z) : jointOriginalPos;
            joint.localPosition = new Vector3(currentBase.x + Mathf.Sin(timer) * bobAmount.x, currentBase.y + Mathf.Sin(timer) * bobAmount.y, currentBase.z + Mathf.Sin(timer) * bobAmount.z) + shakeOffset;
        }
        else
        {
            timer = 0;
            Vector3 targetBase = isCrouched ? new Vector3(jointOriginalPos.x, jointOriginalPos.y * crouchHeight, jointOriginalPos.z) : jointOriginalPos;
            joint.localPosition = new Vector3(Mathf.Lerp(joint.localPosition.x, targetBase.x, Time.deltaTime * bobSpeed), Mathf.Lerp(joint.localPosition.y, targetBase.y, Time.deltaTime * bobSpeed), Mathf.Lerp(joint.localPosition.z, targetBase.z, Time.deltaTime * bobSpeed)) + shakeOffset;
        }
    }
}
